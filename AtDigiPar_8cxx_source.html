393 TString AtDigiPar::GetFile(Int_t fileNum)
394 {
395  std::ifstream fileList;
396  TString sysFile = gSystem->Getenv("VMCWORKDIR");
397  TString parFile = sysFile + "/parameters/AT.files.par";
398  fileList.open(parFile.Data());
399 
400  if (!fileList) {
401  LOG(fatal) << "File " << parFile.Data() << " not found!";
402  throw;
403  }
404 
405  std::string buffer;
406  for (Int_t iFileNum = 0; iFileNum < fileNum + 1; ++iFileNum) {
407  if (fileList.eof()) {
408  LOG(fatal) << "Did not find string #" << fileNum << " in file " << parFile;
409  throw;
410  }
411  std::getline(fileList, buffer);
412  }
413 
414  fileList.close();
415 
416  return TString(sysFile + "/" + buffer);

658 Double_t AtLmedsMod::Fit3D(vector<int> inliners, TVector3 &V1, TVector3 &V2)
659 {
660 
661  //------3D Line Regression
662  //----- adapted from: http://fr.scribd.com/doc/31477970/Regressions-et-trajectoires-3D
663  int R, C;
664  double Q;
665  double Xm, Ym, Zm;
666  double Xh, Yh, Zh;
667  double a, b;
668  double Sxx, Sxy, Syy, Sxz, Szz, Syz;
669  double theta;
670  double K11, K22, K12, K10, K01, K00;
671  double c0, c1, c2;
672  double p, q, r, dm2;
673  double rho, phi;
674 
675  Q = Xm = Ym = Zm = 0.;
676  double total_charge = 0;
677  Sxx = Syy = Szz = Sxy = Sxz = Syz = 0.;
678 
679  for (auto i : inliners) {
680  Q += vQ[i] / 10.;
681  Xm += vX[i] * vQ[i] / 10.;
682  Ym += vY[i] * vQ[i] / 10.;
683  Zm += vZ[i] * vQ[i] / 10.;
684  Sxx += vX[i] * vX[i] * vQ[i] / 10.;
685  Syy += vY[i] * vY[i] * vQ[i] / 10.;
686  Szz += vZ[i] * vZ[i] * vQ[i] / 10.;
687  Sxy += vX[i] * vY[i] * vQ[i] / 10.;
688  Sxz += vX[i] * vZ[i] * vQ[i] / 10.;
689  Syz += vY[i] * vZ[i] * vQ[i] / 10.;
690  }
691  // vTrackCharge.push_back(total_charge);
692 
693  Xm /= Q;
694  Ym /= Q;
695  Zm /= Q;
696  Sxx /= Q;
697  Syy /= Q;
698  Szz /= Q;
699  Sxy /= Q;
700  Sxz /= Q;
701  Syz /= Q;
702  Sxx -= (Xm * Xm);
703  Syy -= (Ym * Ym);
704  Szz -= (Zm * Zm);
705  Sxy -= (Xm * Ym);
706  Sxz -= (Xm * Zm);
707  Syz -= (Ym * Zm);
708 
709  theta = 0.5 * atan((2. * Sxy) / (Sxx - Syy));
710 
711  K11 = (Syy + Szz) * pow(cos(theta), 2) + (Sxx + Szz) * pow(sin(theta), 2) - 2. * Sxy * cos(theta) * sin(theta);
712  K22 = (Syy + Szz) * pow(sin(theta), 2) + (Sxx + Szz) * pow(cos(theta), 2) + 2. * Sxy * cos(theta) * sin(theta);
713  // K12 = -Sxy * (pow(cos(theta), 2) - pow(sin(theta), 2)) + (Sxx - Syy) * cos(theta) * sin(theta);
714  K10 = Sxz * cos(theta) + Syz * sin(theta);
715  K01 = -Sxz * sin(theta) + Syz * cos(theta);
716  K00 = Sxx + Syy;
717 
718  c2 = -K00 - K11 - K22;
719  c1 = K00 * K11 + K00 * K22 + K11 * K22 - K01 * K01 - K10 * K10;
720  c0 = K01 * K01 * K11 + K10 * K10 * K22 - K00 * K11 * K22;
721 
722  p = c1 - pow(c2, 2) / 3.;
723  q = 2. * pow(c2, 3) / 27. - c1 * c2 / 3. + c0;
724  r = pow(q / 2., 2) + pow(p, 3) / 27.;
725 
726  if (r > 0)
727  dm2 = -c2 / 3. + pow(-q / 2. + sqrt(r), 1. / 3.) + pow(-q / 2. - sqrt(r), 1. / 3.);
728  else {
729  rho = sqrt(-pow(p, 3) / 27.);
730  phi = acos(-q / (2. * rho));
731  dm2 = min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos(phi / 3.),
732  min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 2. * TMath::Pi()) / 3.),
733  -c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 4. * TMath::Pi()) / 3.)));
734  }
735 
736  a = -K10 * cos(theta) / (K11 - dm2) + K01 * sin(theta) / (K22 - dm2);
737  b = -K10 * sin(theta) / (K11 - dm2) - K01 * cos(theta) / (K22 - dm2);
738 
739  Xh = ((1. + b * b) * Xm - a * b * Ym + a * Zm) / (1. + a * a + b * b);
740  Yh = ((1. + a * a) * Ym - a * b * Xm + b * Zm) / (1. + a * a + b * b);
741  Zh = ((a * a + b * b) * Zm + a * Xm + b * Ym) / (1. + a * a + b * b);
742 
743  V1.SetXYZ(Xm, Ym, Zm);
744  V2.SetXYZ(Xh, Yh, Zh);
745 
746  return (fabs(dm2 / Q));
747 }

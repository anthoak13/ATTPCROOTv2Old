134 void AtPSA::CalcLorentzVector()

136

140  Double_t TiltRad = fTiltAng * TMath::Pi() / 180.0;

143  Double_t y = front * ot * ot * TMath::Cos(TiltRad) * TMath::Sin(TiltRad);
144  Double_t z = front * (1 + ot * ot * TMath::Cos(TiltRad) * TMath::Cos(TiltRad));

146  fLorentzVector.SetXYZ(x, y, z);

53 void AtPulseTask::setParameters()
54 {
55  fGain = fPar->GetGain();
56  fGETGain = fPar->GetGETGain(); // Get the electronics gain in fC
57  fGETGain = 1.602e-19 * 4096 / (fGETGain * 1e-15); // Scale to gain and correct for ADC
58  fPeakingTime = fPar->GetPeakingTime() / 1000.;
59  fTBTime = fPar->GetTBTime() / 1000.; // in us
60  fNumTbs = fPar->GetNumTbs();
61 
62  fTBEntrance = fPar->GetTBEntrance();
63  fTBPadPlane = fTBEntrance - fPar->GetZPadPlane() / 10. / fTBTime / fPar->GetDriftVelocity();
64 
65  // GET gain: 120fC, 240fC, 1pC or 10pC in 4096 channels
66  // 1.6e-19*4096/120e-15 = 5.4613e-03
67  // 1.6e-19*4096/240e-15 = 2.731e-03
68  // 1.6e-19*4096/1e-12 = 6.5536e-04
69  // 1.6e-19*4096/10e-12 = 6.5536e-05
70 
71  std::cout << " Gain in AtTPC gas: " << fGain << std::endl;
72  std::cout << " GET Gain: " << fGETGain << std::endl;
73  std::cout << " Electronic peaking time: " << fPeakingTime << " us" << std::endl;
74  std::cout << " Number of pads: " << fMap->GetNumPads() << std::endl;
75  std::cout << " Window at TB: " << fTBEntrance << std::endl;
76  std::cout << " Pad plane at TB: " << fTBPadPlane << std::endl;
77 
78  gain = std::make_unique<TF1>("gain",
79  "pow([1]+1,[1]+1)/ROOT::Math::tgamma([1]+1)*pow((x/[0]),[1])*exp(-([1]+1)*(x/[0]))", 0,
80  fGain * 5); // Polya distribution of gain
81  gain->SetParameter(0, fGain);
82  gain->SetParameter(1, 1);
83 
84  auto b = gain->GetParameter(1);
85  avgGainDeviation = fGain / (b + 1);
86  avgGainDeviation *=
87  TMath::Sqrt(TMath::Gamma(b + 3) / TMath::Gamma(b + 1) -
88  TMath::Gamma(b + 2) * TMath::Gamma(b + 2) / TMath::Gamma(b + 1) / TMath::Gamma(b + 1));
89 
90  std::cout << " GET gain deviation: " << avgGainDeviation << std::endl;
91 }

139 void AtPulseTask::saveMCInfo(int mcPointID, int padNumber, int trackID)
140 {
141  // Count occurrences of simPoints coming from the same mcPoint
142  int count = 0;
143 
144  // The same MC point ID is saved per pad only once, but duplicates are allowed in other pads
145  for (auto it = MCPointsMap.lower_bound(padNumber); it != MCPointsMap.upper_bound(padNumber); ++it) {
146  auto mcPointMap = dynamic_cast<AtMCPoint *>(fMCPointArray->At(mcPointID));
147  auto trackIDMap = mcPointMap->GetTrackID();
148  if (it->second == mcPointID || (trackID == trackIDMap)) {
149  ++count;

152  }
153 
154  // insert if the mcPointID and trackID do not both match any existing point
155  if (count == 0)
156  MCPointsMap.insert(std::make_pair(padNumber, mcPointID));

172 bool AtPulseTask::gatherElectronsFromSimulatedPoint(AtSimulatedPoint *point)
173 {
174  if (point == nullptr)
175  return false;
176 
177  auto coord = point->GetPosition();
178  auto xElectron = coord.x(); // mm
179  auto yElectron = coord.y(); // mm
180  auto eTime = coord.z(); // us
181  eTime += fTBPadPlane * fTBTime; // correct time for pad plane location
182  auto charge = point->GetCharge(); // number of electrons
183 
184  auto binNumber = fPadPlane->Fill(xElectron, yElectron);
185  auto padNumber = fMap->BinToPad(binNumber);
186  auto mcPoint = dynamic_cast<AtMCPoint *>(fMCPointArray->At(point->GetMCPointID()));
187  auto trackID = mcPoint->GetTrackID();
188 
189  if (padNumber < 0 || padNumber > fMap->GetNumPads()) {
190  LOG(debug) << "Skipping electron...";
191  return false;
192  }
193 
194  if (fIsSaveMCInfo)
195  saveMCInfo(point->GetMCPointID(), padNumber, trackID);
196 
197  auto totalyInhibited = fMap->IsInhibited(padNumber) == AtMap::kTotal;
198  if (!totalyInhibited) {
199  eleAccumulated[padNumber]->Fill(eTime, charge);
200  electronsMap[padNumber] = eleAccumulated[padNumber].get();
201  }
202 
203  return true;
204 }

231 void AtPulseTask::generateTracesFromGatheredElectrons()
232 {
233  TAxis *axis = eleAccumulated[0]->GetXaxis();
234  Double_t binWidth = axis->GetBinWidth(10);
235 
236  std::vector<Int_t> signal(fNumTbs, 0);
237  for (auto &ite2 : electronsMap) {
238  for (Int_t kk = 0; kk < fNumTbs; kk++)
239  signal[kk] = 0;
240  Int_t thePadNumber = (ite2.first);
241 
242  for (Int_t kk = 0; kk < fNumTbs; kk++) {
243  if (eleAccumulated[thePadNumber]->GetBinContent(kk) > 0) {
244  for (Int_t nn = kk; nn < fNumTbs; nn++) {
245  Double_t binCenter = axis->GetBinCenter(kk);
246  Double_t factor = (((((Double_t)nn) + 0.5) * binWidth) - binCenter) / fPeakingTime;
247  signal[nn] += eleAccumulated[thePadNumber]->GetBinContent(kk) * pow(2.718, -3 * factor) * sin(factor) *
248  pow(factor, 3);
249  }
250  }
251  }
252 
253  // Create pad
254  auto pad = fRawEvent->AddPad(thePadNumber);
255 
256  auto PadCenterCoord = fMap->CalcPadCenter(thePadNumber);
257  pad->SetValidPad(kTRUE);
258  pad->SetPadCoord(PadCenterCoord);
259  pad->SetPedestalSubtracted(kTRUE);
260 
261  auto gAvg = getAvgGETgain(eleAccumulated[thePadNumber]->GetEntries());
262  auto lowGain = fMap->IsInhibited(thePadNumber) == AtMap::kLowGain ? fLowGainFactor : 1;
263 
264  for (Int_t bin = 0; bin < fNumTbs; bin++) {
265  pad->SetADC(bin, signal[bin] * gAvg * fGETGain * lowGain); // NOLINT
266  }
267  }
268 
269  // if electronsMap.size==0, fEventID still needs to be set
270  fRawEvent->SetEventID(fEventID);
271  fRawEvent->SetSimMCPointMap(MCPointsMap);
272  fRawEvent->SetIsGood(true);
273 
274  std::cout << "AtPulseTask Event ID : " << fEventID << "\n";
275  ++fEventID;
276 }

278 double AtPulseTask::getAvgGETgain(Int_t numElectrons)
279 {
280  Double_t gAvg = 0;
281  if (fUseFastGain && numElectrons > 10)
282  gAvg = gRandom->Gaus(fGain, avgGainDeviation / TMath::Sqrt(numElectrons));
283  else {
284  for (Int_t i = 0; i < numElectrons; i++)
285  gAvg += gain->GetRandom();
286  if (numElectrons > 0)

288  }

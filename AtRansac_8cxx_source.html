372 void AtRANSACN::AtRansac::FindVertex(std::vector<AtTrack *> tracks)
373 {
374 
375  // Assumes the minimum distance between two lines, with respect a given threshold, the first vertex candidate. Then
376  // evaluates the distance of each remaining line with respect to the others (vertex) to decide the particles of the
377  // reaction.
378  // std::cout<<" New find vertex call "<<std::endl;
379 
380  Double_t mad = 10; // Minimum approach distance. This is the minimum distance between the lines in 3D. Must be bigger
381  // than fLineDistThreshold
382  XYZVector c_1(-1000, -1000, -1000);
383  XYZVector c_2(-1000, -1000, -1000);
384  // std::vector<AtTrack*> *TrackCand;
385 
386  // Current parametrization
387  // x = p[0] + p[1]*t;
388  // y = p[2] + p[3]*t;
389  // z = t;
390  // (x,y,z) = (p[0],p[2],0) + (p[1],p[3],1)*t
391 
392  // Equation of the Z axis in the solenoid frame
393  XYZVector Z_0(0, 0, 1000.0);
394  XYZVector Z_1(0, 0, 1);
395 
396  // Equation of the Y axis in the
397  XYZVector Y_0(0, 0, 1000.0);
398  XYZVector Y_1(0, 1, 0);
399 
400  // Counter clockwise rotations
401  // Direction of the Z and Y axis in the detector frame (This is the rotated detector), this is used to calculater the
402  // theta and phi angles
403  RotationX rx(-fTiltAng * TMath::Pi() / 180.0);
404  // XYZVector Z_1_rot(rx * Z_1);
405 
406  RotationX ry(-fTiltAng * TMath::Pi() / 180.0);
407  // XYZVector Y_1_rot(ry * Y_1);
408 
409  // Vector of the beam determined from the experimental data
410  // XYZVector BeamDir_1(-0.106359,-0.0348344,1.0);
411  XYZVector BeamDir_1(0., 0., 1.0);
412  // TODO:: This is for 6.5 degrees of tilting angle. Need a function to set it.
413 
414  // Test each line against the others to find a vertex candidate
415  for (Int_t i = 0; i < int(tracks.size()) - 1; i++) {
416 
417  AtTrack *track = tracks.at(i);
418  track->SetTrackID(i);
419  std::vector<Double_t> p = track->GetFitPar();
420 
421  if (p.size() > 0) {
422  // XYZVector L_0(p[0], p[2], 0. );//p1
423  // XYZVector L_1(p[1], p[3], 1. );//d1
424  XYZVector L_0(p[0], p[2], p[4]); // p1
425  XYZVector L_1(p[1], p[3], p[5]); // d1
426 
427  // std::cout<<" L_1 p[0] : "<<p[0]<<" L_1 p[2] : "<<p[2]<<std::endl;
428  // std::cout<<" L_1 p[1] : "<<p[1]<<" L_1 p[3] : "<<p[3]<<std::endl;
429 
430  for (Int_t j = i + 1; j < tracks.size(); j++) {
431  AtTrack *track_f = tracks.at(j);
432  track_f->SetTrackID(j);
433  std::vector<Double_t> p_f = track_f->GetFitPar();
434 
435  if (p_f.size() > 0) {
436  // XYZVector L_f0(p_f[0], p_f[2], 0. );//p2
437  // XYZVector L_f1(p_f[1], p_f[3], 1. );//d2
438  XYZVector L_f0(p_f[0], p_f[2], p_f[4]); // p2
439  XYZVector L_f1(p_f[1], p_f[3], p_f[5]); // d2
440 
441  // std::cout<<" L_f0 p_f[0] : "<<p_f[0]<<" L_f1 p_f[2] : "<<p_f[2]<<std::endl;
442  // std::cout<<" L_f1 p_f[1] : "<<p_f[1]<<" L_f1 p_f[3] : "<<p_f[3]<<std::endl;
443 
444  XYZVector L = L_1.Cross(L_f1);
445  Double_t L_mag = L.Rho();
446  XYZVector n_1 = L_1.Cross(L);
447  XYZVector n_2 = L_f1.Cross(L);
448  c_1 = L_0 + ((L_f0 - L_0).Dot(n_2) * L_1) / (L_1.Dot(n_2));
449  c_2 = L_f0 + ((L_0 - L_f0).Dot(n_1) * L_f1) / (L_f1.Dot(n_1));
450 
451  // std::cout<<i<<" "<<j<<" "<<L_mag<<std::endl;
452 
453  if (L_mag > 0) {
454  XYZVector n = L / (Double_t)L_mag;
455  Double_t d = TMath::Abs(n.Dot(L_0 - L_f0));
456  // std::cout<<" Distance of minimum approach : "<<d<<std::endl;
457  Double_t num = L_1.X() * L_f1.X() + L_1.Y() * L_f1.Y() + L_1.Z() * L_f1.Z();
458  Double_t den = TMath::Sqrt(L_1.X() * L_1.X() + L_1.Y() * L_1.Y() + L_1.Z() * L_1.Z()) *
459  TMath::Sqrt(L_f1.X() * L_f1.X() + L_f1.Y() * L_f1.Y() + L_f1.Z() * L_f1.Z());
460  // NB:: The vector of hits is always sorted in time so the direction of porpagation of the particle is
461  // always positive. This means that the angle will be always between 0 and 90 degree. The vertex and
462  // the mean time of the track are needed to determine the direction of the track and then add 90
463  // degrees.
464  Double_t ang2 = TMath::ACos(num / den);
465  XYZVector vertex1_buff;
466  vertex1_buff.SetXYZ(c_1.X(), c_1.Y(), c_1.Z());
467  XYZVector vertex2_buff;
468  vertex2_buff.SetXYZ(c_2.X(), c_2.Y(), c_2.Z());
469 
470  // Angle with respect to solenoid
471  Double_t angZi = GetAngleTracks(L_1, Z_1);
472  Double_t angZj = GetAngleTracks(L_f1, Z_1);
473 
474  // Angles with respect to tilted detector
475  Double_t angZDeti = GetAngleTracks(L_1, BeamDir_1);
476  Double_t angZDetj = GetAngleTracks(L_f1, BeamDir_1);
477  Double_t angYDeti = 0.0; // GetAngleTracks(L_1,Y_1_rot);
478  Double_t angYDetj = 0.0; // GetAngleTracks(L_f1,Y_1_rot);
479 
480  track->SetAngleZAxis(angZi);
481  track->SetAngleZDet(angZDeti);
482  track->SetAngleYDet(angYDeti);
483 
484  track_f->SetAngleZAxis(angZj);
485  track_f->SetAngleZDet(angZDetj);
486  track_f->SetAngleYDet(angYDetj);
487 
488  XYZPoint vertexAvg(0.5 * (vertex1_buff + vertex2_buff));
489  track->SetTrackVertex(vertexAvg);
490  track_f->SetTrackVertex(vertexAvg);
491 
492  if (d < mad) {
493 
494  mad = d;
495  // std::cout<<" New distance of minimum approach : "<<mad<<std::endl;
496  // std::cout<<" Angle between lines i : "<<i<<" j : "<<j<<" "<<ang2<<std::endl;
497 
498  // Global event variables
499  fVertex_1.SetXYZ(c_1.X(), c_1.Y(), c_1.Z());
500  fVertex_2.SetXYZ(c_2.X(), c_2.Y(), c_2.Z());
501  fVertex_mean = 0.5 * (fVertex_1 + fVertex_2);
502  fVertex_tracks.first = i;
503  fVertex_tracks.second = j;
504  fMinimum = mad;
505 
506  if (!CheckTrackID(track->GetTrackID(), &fTrackCand)) {
507  fTrackCand.push_back(*track);
508  PairedLines PL;
509  PL.LinesID.first = i;
510  PL.LinesID.second = j;
511  PL.AngleZAxis.first = angZi;
512  PL.AngleZAxis.second = angZj;
513  PL.AngleZDet.first = angZDeti;
514  PL.AngleZDet.second = angZDetj;
515  PL.AngleYDet.first = angYDeti;
516  PL.AngleYDet.second = angYDetj;
517  PL.minDist = mad;
518  PL.meanVertex = fVertex_mean;
519  PL.angle = ang2;
520  PLines.push_back(PL);
521  }
522  if (!CheckTrackID(track_f->GetTrackID(), &fTrackCand)) {
523  fTrackCand.push_back(*track_f);
524  PairedLines PL;
525  PL.LinesID.first = i;
526  PL.LinesID.second = j;
527  PL.AngleZAxis.first = angZi;
528  PL.AngleZAxis.second = angZj;
529  PL.AngleZDet.first = angZDeti;
530  PL.AngleZDet.second = angZDetj;
531  PL.AngleYDet.first = angYDeti;
532  PL.AngleYDet.second = angYDetj;
533  PL.minDist = mad;
534  PL.meanVertex = fVertex_mean;
535  PL.angle = ang2;
536  PLines.push_back(PL);
537  }
538  }
539 
540  if (d < fLineDistThreshold) {
541

590 Double_t AtRANSACN::AtRansac::distance2(double x, double y, double z, const double *p)
591 {
592 
593  // distance line point is D= | (xp-x0) cross ux |
594  // where ux is direction of line and x0 is a point in the line (like t = 0) and x1 is in t=1
595  XYZVector xp(x, y, z);
596  XYZVector x0(p[0], p[2], 0.);
597  XYZVector x1(p[0] + p[1], p[2] + p[3], 1.);
598  XYZVector u = (x1 - x0).Unit();
599  double d2 = ((xp - x0).Cross(u)).Mag2();
600  return d2;
601 }

624  // positive. This means that the angle will be always between 0 and 90 degree. The vertex and the mean time of the

630 
631  return ang;

645 Double_t AtRANSACN::AtRansac::Fit3D(AtTrack *track)
646 {
647 
648  //--------Adapted from https://www.scribd.com/doc/31477970/Regressions-et-trajectoires-3D
649  //--------3D linear regression
650 
651  int R, C;
652  double Q;
653  double Xm, Ym, Zm;
654  double Xh, Yh, Zh;
655  double a, b;
656  double Sxx, Sxy, Syy, Sxz, Szz, Syz;
657  double theta;
658  double K11, K22, K12, K10, K01, K00;
659  double c0, c1, c2;
660  double p, q, r, dm2;
661  double rho, phi;
662 
663  Q = Xm = Ym = Zm = 0.;
664  Sxx = Syy = Szz = Sxy = Sxz = Syz = 0.;
665 
666  std::vector<AtHit> HitArray = track->GetHitArray();
667  std::vector<double> X;
668  std::vector<double> Y;
669  std::vector<double> Z;
670  std::vector<double> Charge;
671 
672  for (auto hit : HitArray) {
673  auto pos = hit.GetPosition();
674  // Double_t tq = hit.GetCharge();
675  X.push_back(pos.X());
676  Y.push_back(pos.Y());
677  Z.push_back(pos.Z());
678  Charge.push_back(q);
679  }
680 
681  for (Int_t i = 0; i < HitArray.size(); i++) {
682  Q += Charge[i] / 10.;
683  Xm += X[i] * Charge[i] / 10.;
684  Ym += Y[i] * Charge[i] / 10.;
685  Zm += Z[i] * Charge[i] / 10.;
686  Sxx += X[i] * X[i] * Charge[i] / 10.;
687  Syy += Y[i] * Y[i] * Charge[i] / 10.;
688  Szz += Z[i] * Z[i] * Charge[i] / 10.;
689  Sxy += X[i] * Y[i] * Charge[i] / 10.;
690  Sxz += X[i] * Z[i] * Charge[i] / 10.;
691  Syz += Y[i] * Z[i] * Charge[i] / 10.;
692  }
693 
694  Xm /= Q;
695  Ym /= Q;
696  Zm /= Q;
697  Sxx /= Q;
698  Syy /= Q;
699  Szz /= Q;
700  Sxy /= Q;
701  Sxz /= Q;
702  Syz /= Q;
703  Sxx -= (Xm * Xm);
704  Syy -= (Ym * Ym);
705  Szz -= (Zm * Zm);
706  Sxy -= (Xm * Ym);
707  Sxz -= (Xm * Zm);
708  Syz -= (Ym * Zm);
709 
710  theta = 0.5 * atan((2. * Sxy) / (Sxx - Syy));
711 
712  K11 = (Syy + Szz) * pow(cos(theta), 2) + (Sxx + Szz) * pow(sin(theta), 2) - 2. * Sxy * cos(theta) * sin(theta);
713  K22 = (Syy + Szz) * pow(sin(theta), 2) + (Sxx + Szz) * pow(cos(theta), 2) + 2. * Sxy * cos(theta) * sin(theta);
714  // K12 = -Sxy * (pow(cos(theta), 2) - pow(sin(theta), 2)) + (Sxx - Syy) * cos(theta) * sin(theta);
715  K10 = Sxz * cos(theta) + Syz * sin(theta);
716  K01 = -Sxz * sin(theta) + Syz * cos(theta);
717  K00 = Sxx + Syy;
718 
719  c2 = -K00 - K11 - K22;
720  c1 = K00 * K11 + K00 * K22 + K11 * K22 - K01 * K01 - K10 * K10;
721  c0 = K01 * K01 * K11 + K10 * K10 * K22 - K00 * K11 * K22;
722 
723  p = c1 - pow(c2, 2) / 3.;
724  q = 2. * pow(c2, 3) / 27. - c1 * c2 / 3. + c0;
725  r = pow(q / 2., 2) + pow(p, 3) / 27.;
726 
727  if (r > 0)
728  dm2 = -c2 / 3. + pow(-q / 2. + sqrt(r), 1. / 3.) + pow(-q / 2. - sqrt(r), 1. / 3.);
729  else {
730  rho = sqrt(-pow(p, 3) / 27.);
731  phi = acos(-q / (2. * rho));
732  dm2 = std::min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos(phi / 3.),
733  std::min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 2. * TMath::Pi()) / 3.),
734  -c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 4. * TMath::Pi()) / 3.)));
735  }
736 
737  a = -K10 * cos(theta) / (K11 - dm2) + K01 * sin(theta) / (K22 - dm2);
738  b = -K10 * sin(theta) / (K11 - dm2) - K01 * cos(theta) / (K22 - dm2);
739 
740  Xh = ((1. + b * b) * Xm - a * b * Ym + a * Zm) / (1. + a * a + b * b);
741  Yh = ((1. + a * a) * Ym - a * b * Xm + b * Zm) / (1. + a * a + b * b);
742  Zh = ((a * a + b * b) * Zm + a * Xm + b * Ym) / (1. + a * a + b * b);
743 
744  std::vector<Double_t> par;
745  par.push_back(Xm); // 0
746  par.push_back(Xh - Xm); // 1
747  par.push_back(Ym); // 2
748  par.push_back(Yh - Ym); // 3
749  par.push_back(Zm); // 4
750  par.push_back(Zh - Zm); // 5
751 
752  track->SetFitPar(par);
753  track->SetMinimum(fabs(dm2 / Q));
754  track->SetNFree(X.size() - 6);
755  X.clear();
756  Y.clear();
757  Z.clear();
758  Charge.clear();
759 
760  return (fabs(dm2 / Q));
761 }
